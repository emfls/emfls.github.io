복구 방법 단계별 가이드

  2. PowerShell 복원: 백업한 CSV 파일을 이용한 일괄 복원

  5. 레지스트리 복원: 백업한 레지스트리 파일 복원
# PowerShell 일괄 복원 스크립트
# CSV 백업 파일에서 서비스 상태 복원
$BackupFile = "C:\Backup\Services_Backup.csv"
$Services = Import-Csv -Path $BackupFile

foreach ($Service in $Services) {
try {
Set-Service -Name $Service.Name -StartupType $Service.StartType
if ($Service.Status -eq "Running") {
Start-Service -Name $Service.Name
}
Write-Host "Restored: $($Service.Name)"
} catch {
Write-Warning "Failed to restore: $($Service.Name)"
}
}
